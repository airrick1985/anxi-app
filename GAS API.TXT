/**
 * 一個強大的日誌函數，將資訊寫入到指定的 Sheet 中。
 * @param {*} message - 你想要記錄的主要訊息。
 * @param {*} [details={}] - (可選) 一個物件，包含你想看的變數或詳細資訊。
 */
function writeToSheetLog(message, details = {}) {
  try {
    const sheet = SpreadsheetApp.openById(LOG_SHEET_ID).getSheetByName('日誌'); // 假設工作表叫「日誌」
    if (!sheet) return; // 如果找不到，就靜默失敗

    const timestamp = new Date();
    const detailsString = JSON.stringify(details, null, 2); // 格式化 JSON
    
    sheet.appendRow([timestamp, message, detailsString]);
  } catch (e) {
    // 如果連寫日誌都出錯，就在內建日誌中記錄下來
    Logger.log(`[writeToSheetLog FAILED] ${e.toString()}`);
  }
}

function doPost(e) {
  let payload;
  try {
    payload = JSON.parse(e.postData.contents);
    Logger.log('[doPost] 接收到 payload: ' + JSON.stringify(payload));
  } catch (error) {
    Logger.log('[doPost] ❌ JSON 解析錯誤: ' + e.postData.contents);
    return createJsonResponse({ status: 'error', message: '無效的請求資料' });
  }

  const action = payload.action;
  const projectNameFromPayload = payload.projectName;
  Logger.log('[doPost] Action: ' + action);
  Logger.log('[doPost] ProjectName from payload: ' + projectNameFromPayload);

  let ssId = null;

  const actionsRequiringSsId = [
    'get_unit_list', 'get_building_list', 'get_house_detail', 'get_inspection_records',
    'get_all_house_details', 'update_inspection_record', 'get_repair_status_options',
    'add_inspection_record', 'get_dropdown_options', 'edit_inspection_record',
    'delete_inspection_record', 'get_all_subcategories', 'get_deleted_inspection_records',
    'restore_inspection_record', 'delete_photo_from_record', 'generate_share_url',
    'get_shared_inspection_records', 'edit_inspection_record_with_photos',
    'confirm_inspection', 'generate_inspection_pdf', 'get_all_project_inspection_records'
    // 確保這裡列出了所有需要 ssId 的 action
  ];

  if (actionsRequiringSsId.includes(action)) {
    if (!projectNameFromPayload) {
      Logger.log('[doPost] ❌ Action "' + action + '" 需要 projectName，但 payload 中未提供。');
      return createJsonResponse({ status: 'error', message: '請求缺少建案名稱 (projectName) 參數。' });
    }
    try {
      Logger.log('[doPost] 嘗試為建案 "' + projectNameFromPayload + '" 獲取 ssId...');
      ssId = getSpreadsheetIdByProjectName(projectNameFromPayload);
      Logger.log('[doPost] ✅ 獲取到的 ssId: "' + ssId + '" (類型: ' + typeof ssId + ') 對於建案: "' + projectNameFromPayload + '"');

      if (!ssId || String(ssId).trim() === "") {
        Logger.log('[doPost] ❌ 獲取的 ssId 為空或無效，對於建案: ' + projectNameFromPayload);
        return createJsonResponse({ status: 'error', message: `系統錯誤：找不到建案 "${projectNameFromPayload}" 的配置或配置為空。` });
      }
    } catch (err) {
      Logger.log('[doPost] ❌ 獲取 ssId 時發生錯誤，建案: ' + projectNameFromPayload + '. 錯誤: ' + err.message + ' Stack: ' + err.stack);
      return createJsonResponse({ status: 'error', message: `獲取建案 "${projectNameFromPayload}" 資料時發生錯誤: ${err.message}` });
    }
  }

  Object.keys(payload).forEach(key => {
    Logger.log(`[doPost] payload key-value: ${key} = ${payload[key]}`);
  });

  switch (action) {
    case 'login': return handleLogin(payload);
    case 'get_project_list': return handleGetProjectList(payload);
    case 'forgot_password': return handleForgotPassword(payload);
    case 'get_unit_list': return handleGetUnitList(ssId);
    case 'get_building_list': return handleGetBuildingList(payload, ssId);
    case 'get_house_detail': return handleGetHouseDetail(payload, ssId);
    case 'update_profile': return handleUpdateProfile(payload);
    case 'get_all_project_inspection_records': return handleGetAllInspectionRecordsForProject(ssId);
    case 'get_inspection_records': return handleGetInspectionRecords(payload, ssId);
    case 'get_all_house_details': return handleGetAllHouseDetails(ssId);
    case 'update_inspection_record': return handleUpdateInspectionRecord(payload, ssId);
    case 'get_repair_status_options': return handleGetRepairStatusOptions(ssId);
    case 'add_inspection_record': return handleAddInspectionRecord(payload, ssId);
    case 'upload_photo': return handleUploadPhoto(payload); // 假設 upload_photo 不需要 ssId
    case 'get_dropdown_options': return handleGetDropdownOptions(ssId);
    case 'edit_inspection_record': return handleEditInspectionRecord(payload, ssId);
    case 'delete_inspection_record': return handleDeleteInspectionRecord(payload, ssId);
    case 'get_all_subcategories': return handleGetAllSubcategories(ssId);
    case 'get_deleted_inspection_records': return handleGetDeletedInspectionRecords(ssId);
    case 'restore_inspection_record': return handleRestoreInspectionRecord(payload, ssId);
    case 'delete_photo_from_record': return handleDeletePhotoFromRecord(payload, ssId);
    case 'generate_share_url': return handleGenerateShareUrl(payload, ssId);
    case 'get_shared_inspection_records': return handleGetSharedInspectionRecords(payload, ssId);
    case 'edit_inspection_record_with_photos': return handleEditInspectionRecordWithPhotos(payload, ssId);
    case 'upload_signature': return handleUploadSignature(payload); // 假設 upload_signature 不需要 ssId
    case 'confirm_inspection': return handleConfirmInspection(payload, ssId);
    case 'generate_inspection_pdf': return handleGenerateInspectionPdf(payload, ssId);
    case 'get_projects_by_system_permission':return handleGetProjectsBySystemPermission(payload);
    case 'get_sales_control_data':return handleGetSalesControlData(payload);
    case 'get_parking_list':return handleGetParkingList(payload);
    case 'get_quote_personnel_list':return handleGetQuotePersonnelList(payload);
    case 'generate_quote_pdf': return handleGenerateQuotePdf(payload); 
    case 'update_sales_data': return handleUpdateSalesData(payload); 

    default:
      Logger.log('[doPost] ❌ 未知的 action: ' + action);

  }
}

function createJsonResponse(obj) { // 確保你有這個函數
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}


function getSpreadsheetIdByProjectName(projectName) {
  Logger.log('[getSpreadsheetIdByProjectName] 正在查找建案: "' + projectName + '"');
  try {
    const sheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('建案選單');
    if (!sheet) {
      Logger.log('[getSpreadsheetIdByProjectName] ❌ 找不到 "建案選單" 工作表。');
      throw new Error('找不到 "建案選單" 工作表配置。');
    }
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const nameIndex = headers.indexOf('建案名稱');
    const idIndex = headers.indexOf('SheetID');

    if (nameIndex === -1 || idIndex === -1) {
        Logger.log(`[getSpreadsheetIdByProjectName] ❌ "建案選單" 表頭配置錯誤。nameIndex: ${nameIndex}, idIndex: ${idIndex}`);
        throw new Error('"建案選單" 表頭配置不正確 (找不到 "建案名稱" 或 "SheetID")。');
    }

    const lastRow = sheet.getLastRow();
    if (lastRow < 2) { // 沒有數據行
        Logger.log('[getSpreadsheetIdByProjectName] ❌ "建案選單" 工作表中沒有數據。');
        throw new Error(`在 "建案選單" 中找不到建案名稱：${projectName} (無數據)。`);
    }

    const nameColumn = sheet.getRange(2, nameIndex + 1, lastRow - 1).getValues().flat();
    const idColumn = sheet.getRange(2, idIndex + 1, lastRow - 1).getValues().flat();

    for (let i = 0; i < nameColumn.length; i++) {
      Logger.log('[getSpreadsheetIdByProjectName] 比較: "' + nameColumn[i] + '" 與 "' + projectName + '"');
      if (nameColumn[i] === projectName) {
        const foundSsId = idColumn[i];
        Logger.log('[getSpreadsheetIdByProjectName] ✅ 找到匹配! 建案: "' + projectName + '", SheetID: "' + foundSsId + '"');
        if (!foundSsId || String(foundSsId).trim() === "") {
            Logger.log('[getSpreadsheetIdByProjectName] ⚠️ 找到建案 "' + projectName + '" 但其 SheetID 為空。');
            throw new Error(`建案 "${projectName}" 的 SheetID 未設定或為空。`);
        }
        return String(foundSsId).trim(); // 確保返回的是字符串且去除前後空格
      }
    }

    Logger.log('[getSpreadsheetIdByProjectName] ❌ 未找到建案名稱："' + projectName + '" 在 "建案選單" 中。');
    throw new Error(`在 "建案選單" 中找不到建案名稱：${projectName}`);
  } catch (err) {
    Logger.log('[getSpreadsheetIdByProjectName] 內部錯誤: ' + err.message + ' Stack: ' + err.stack);
    throw err;
  }
}





function formatSheetDate(value) {
  const FUNCTION_NAME = 'formatSheetDate'; // 用於日誌
  const scriptTimeZone = Session.getScriptTimeZone();

  if (!value) { // 處理 null, undefined, 空字符串
    return '';
  }

  if (value instanceof Date) {
    try {
      return Utilities.formatDate(value, scriptTimeZone, 'yyyy-MM-dd');
    } catch (e) {
      Logger.log(`[${FUNCTION_NAME}] Error formatting Date object: ${value}. Error: ${e.message}`);
      return value.toString(); // 或返回 ''
    }
  }

  if (typeof value === 'number') {
    // 25569 是 1970/1/1 和 1899/12/30 的天數差 (Excel/Sheets 日期基准)
    // 40000 大約是 2009年，一個合理的下限，避免將小數字誤認為日期
    if (value > 25569 && value < 70000) { // 70000 大約是 2190 年
      try {
        // (value - 25569) * 24 * 60 * 60 * 1000
        return Utilities.formatDate(new Date((value - 25569) * 86400000), scriptTimeZone, 'yyyy-MM-dd');
      } catch (e) {
        Logger.log(`[${FUNCTION_NAME}] Error formatting number ${value} as date. Error: ${e.message}`);
        return value.toString();
      }
    } else {
      // Logger.log(`[${FUNCTION_NAME}] Number ${value} is not in typical date serial range.`);
      return value.toString(); // 不是典型日期序列號，返回原樣
    }
  }

  if (typeof value === 'string') {
    try {
      const dateObject = new Date(value);
      if (!isNaN(dateObject.valueOf())) { // 檢查是否為有效日期
        return Utilities.formatDate(dateObject, scriptTimeZone, 'yyyy-MM-dd');
      } else {
        // Logger.log(`[${FUNCTION_NAME}] Could not parse string "${value}" into a valid Date.`);
        return value; // 如果無法解析，返回原始字符串
      }
    } catch (e) {
      // Logger.log(`[${FUNCTION_NAME}] Exception parsing string "${value}". Error: ${e.message}`);
      return value; // 解析異常，返回原始字符串
    }
  }

  // 對於其他未知類型，返回其字符串表示或空字符串
  Logger.log(`[${FUNCTION_NAME}] Unhandled value type: ${typeof value}, value: ${value}. Returning as string or empty.`);
  return value ? String(value) : '';
}


// GAS
function handleGetAllInspectionRecordsForProject(ssId) {
  Logger.log('[handleGetAllInspectionRecordsForProject] Received ssId: ' + ssId);
  if (!ssId) {
    Logger.log('[handleGetAllInspectionRecordsForProject] ❌ Invalid ssId.');
    return createJsonResponse({ status: 'error', message: 'System error: Cannot identify target sheet (ssId invalid).' });
  }
  try {
    const sheet = SpreadsheetApp.openById(ssId).getSheetByName('驗屋紀錄');
    if (!sheet) {
      Logger.log('[handleGetAllInspectionRecordsForProject] ❌ "驗屋紀錄" sheet not found in ssId: ' + ssId);
      return createJsonResponse({ status: 'error', message: 'System configuration error: Inspection records sheet not found.' });
    }
    const data = sheet.getDataRange().getValues();
    const header = data[0];
    const records = [];

    const col = (name) => header.indexOf(name);
    const deletedColumnName = '已刪除'; // 確認你的欄位名
    const deletedIndex = header.indexOf(deletedColumnName);

    // 建議檢查所有必要的欄位索引是否為 -1

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      // 如果有已刪除欄位，則過濾掉已刪除的
      if (deletedIndex !== -1 && row[deletedIndex] === 'Y') {
        continue;
      }

      records.push({
        createdAt: formatSheetDate(row[col('建檔時間')]),
        key: row[col('KEY')],
        inspectionDate: formatSheetDate(row[col('驗屋日期')]),
        inspectionStage: row[col('驗屋階段')],
        inspector: row[col('驗屋人')],
        owner: row[col('產權人')],
        unit: row[col('戶別')], // 包含戶別信息很重要
        area: row[col('檢查區域')],
        category: row[col('分類')],
        subcategory: row[col('細項')],
        inspectionStatus: row[col('檢查狀態')],
        defectLevel: row[col('缺失等級')],
        description: row[col('檢查說明')],
        repairDate: formatSheetDate(row[col('檢修時間')]),
        repairStatus: row[col('檢修狀態')],
        repairDescription: row[col('檢修說明')],
        photo1: row[col('缺失照片1')],
        photo2: row[col('缺失照片2')],
        photo3: row[col('缺失照片3')],
        photo4: row[col('缺失照片4')],
        defectPhoto1: row[col('檢修照片1')],
        defectPhoto2: row[col('檢修照片2')],
        defectPhoto3: row[col('檢修照片3')],
        defectPhoto4: row[col('檢修照片4')],
        deleted: row[col('已刪除')] || '' // 即使過濾了，也可能想保留這個字段
      });
    }
    Logger.log('[handleGetAllInspectionRecordsForProject] ✅ Found ' + records.length + ' records for project.');
    return createJsonResponse({ status: 'success', records });
  } catch (err) {
    Logger.log('[handleGetAllInspectionRecordsForProject] ❌ Error: ' + err.message + ' Stack: ' + err.stack);
    return createJsonResponse({ status: 'error', message: 'Error fetching all inspection records: ' + err.message });
  }
}

function handleGetSharedInspectionRecords({ unitId, token }, ssId) {
  if (!unitId || !token) {
    return createJsonResponse({ status: 'error', message: '缺少 unitId 或 token' });
  }

  const ss = SpreadsheetApp.openById(ssId);
  const shareSheet = ss.getSheetByName('驗屋分享網址');
  const shareData = shareSheet.getDataRange().getValues();

  let tokenMatched = false;
  for (let i = 1; i < shareData.length; i++) {
    if (shareData[i][0] === unitId && shareData[i][1] === token) {
      tokenMatched = true;
      break;
    }
  }

  if (!tokenMatched) {
    return createJsonResponse({ status: 'error', message: '無效的分享連結' });
  }

  const sheet = ss.getSheetByName('驗屋紀錄');
  const data = sheet.getDataRange().getValues();
  const header = data[0];
  const getIndex = (name) => header.indexOf(name);

const formatDriveUrl = (url) => {
  if (!url) return '';
  const patterns = [
    /\/file\/d\/([a-zA-Z0-9_-]+)\//,  // for /file/d/xxx/view
    /id=([a-zA-Z0-9_-]+)/             // for /open?id=xxx
  ];
  for (const pattern of patterns) {
    const match = url.match(pattern);
    if (match) {
      return `https://drive.google.com/uc?export=view&id=${match[1]}`;
    }
  }
  return '';
};


  const records = [];
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[getIndex('戶別')] !== unitId) continue;
    if (row[getIndex('已刪除')] === 'Y') continue;

    records.push({
      createdAt: formatSheetDate(row[getIndex('建檔時間')]),
      key: row[getIndex('KEY')],
      inspectionDate: formatSheetDate(row[getIndex('驗屋日期')]),
      inspectionStage: row[getIndex('驗屋階段')],
      area: row[getIndex('檢查區域')],
      category: row[getIndex('分類')],
      subcategory: row[getIndex('細項')],
      description: row[getIndex('檢查說明')],
      photo1: formatDriveUrl(row[getIndex('缺失照片1')]),
      photo2: formatDriveUrl(row[getIndex('缺失照片2')]),
      photo3: formatDriveUrl(row[getIndex('缺失照片3')]),
      photo4: formatDriveUrl(row[getIndex('缺失照片4')]),
    });
  }

  return createJsonResponse({ status: 'success', records });
}



function handleAddInspectionRecord(payload, ssId) {
  const sheet = SpreadsheetApp.openById(ssId).getSheetByName('驗屋紀錄');
  const values = [
    payload.createdAt, payload.key, payload.inspectionDate, payload.inspectionStage,

  sheet.appendRow(values);
  return createJsonResponse({ status: 'success', message: '已新增紀錄' });
}

//填寫檢修
function handleUpdateInspectionRecord({ key, repairDate, repairStatus, repairDescription },ssId) {
  const sheet = SpreadsheetApp.openById(ssId).getSheetByName('驗屋紀錄');
  const data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (data[i][1] === key) {
      if (repairDate) sheet.getRange(i + 1, 14).setValue(repairDate);
      sheet.getRange(i + 1, 15).setValue(repairStatus || '');
      sheet.getRange(i + 1, 16).setValue(repairDescription || '');
      return createJsonResponse({ status: 'success' });
    }
  }
  return createJsonResponse({ status: 'error', message: '找不到對應資料 (key 不存在)' });
}

//修改驗屋紀錄
function handleEditInspectionRecord(payload , ssId) {
  const sheet = SpreadsheetApp.openById(ssId).getSheetByName('驗屋紀錄');
  const data = sheet.getDataRange().getValues();

  const { key } = payload;
  if (!key) return createJsonResponse({ status: 'error', message: '缺少 key 參數' });

  for (let i = 1; i < data.length; i++) {
    if (data[i][1] === key) {
      // 依照你現有欄位順序：第 1 欄是 createdAt，第 2 欄是 key，第 3~15 欄依序更新
      sheet.getRange(i + 1, 3).setValue(payload.inspectionDate || '');
      sheet.getRange(i + 1, 4).setValue(payload.inspectionStage || '');
      sheet.getRange(i + 1, 5).setValue(payload.inspector || '');
      sheet.getRange(i + 1, 6).setValue(payload.owner || '');
      sheet.getRange(i + 1, 7).setValue(payload.unit || '');
      sheet.getRange(i + 1, 8).setValue(payload.area || '');
      sheet.getRange(i + 1, 9).setValue(payload.category || '');
      sheet.getRange(i + 1, 10).setValue(payload.subcategory || '');
      sheet.getRange(i + 1, 11).setValue(payload.inspectionStatus || '');
      sheet.getRange(i + 1, 12).setValue(payload.defectLevel || '');
      sheet.getRange(i + 1, 13).setValue(payload.description || '');
      sheet.getRange(i + 1, 14).setValue(payload.repairDate || '');
      sheet.getRange(i + 1, 15).setValue(payload.repairStatus || '');
      sheet.getRange(i + 1, 16).setValue(payload.repairDescription || '');
      return createJsonResponse({ status: 'success', message: '驗屋紀錄已更新' });
    }
  }

  return createJsonResponse({ status: 'error', message: '找不到對應驗屋紀錄（key 不存在）' });
}



function handleUploadPhoto({ filename, base64 }) {
  if (!filename || !base64) return createJsonResponse({ status: 'error', message: '缺少參數' });

  const blob = Utilities.newBlob(Utilities.base64Decode(base64), 'image/jpeg', filename);
  const file = DriveApp.getFolderById('1js6n9EOFaiN1YrPKIgR3-_d-Gc5b8Lma').createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return createJsonResponse({ status: 'success', url: file.getUrl(), name: file.getName(), id: file.getId() });
}

function handleGetAllHouseDetails(ssId) {

function handleLogin(payload) {
  Logger.log("[handleLogin] Handling login for key: " + payload.key + ", projectName: " + payload.projectName);

  const sheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('用戶資料');
  const data = sheet.getDataRange().getValues();
  const header = data[0];

  const phoneIndex = header.indexOf('手機號碼');
  const emailIndex = header.indexOf('EMAIL');
  const nameIndex = header.indexOf('NAME');
  const passwordIndex = header.indexOf('密碼');
  const projectIndex = header.indexOf('建案權限');
  // const permissionsIndex = header.indexOf('功能權限'); // 舊邏輯，不再需要

  let result = { status: 'not_found', message: '手機號碼不存在或錯誤' };
  let loginStatus = '帳號不存在';
  let userName = '';

  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row[phoneIndex] === payload.key) {
      userName = row[nameIndex];
      Logger.log("[handleLogin] Found user: " + userName + " for key: " + payload.key);

      if (row[passwordIndex] !== payload.password) {
        result = { status: 'wrong_password', message: '密碼錯誤' };
        loginStatus = '密碼錯誤';
        Logger.log("[handleLogin] Password mismatch for user: " + userName);
        break;
      }
      Logger.log("[handleLogin] Password matched for user: " + userName);

      // --- 權限新邏輯開始 ---
      // 從 '權限管理' 工作表獲取功能權限
      let userPermissions = [];
      try {
        const permissionSheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('權限管理');
        if (permissionSheet) {
          const permissionData = permissionSheet.getDataRange().getValues();
          if (permissionData.length > 1) { // 至少要有表頭和一筆資料
            const permissionHeader = permissionData[0];
            const pPhoneIndex = permissionHeader.indexOf('手機號碼');
            const pSystemFunctionIndex = permissionHeader.indexOf('系統功能');
            const pPermissionIndex = permissionHeader.indexOf('權限');

            // 確認所有必要的欄位都存在
            if (pPhoneIndex !== -1 && pSystemFunctionIndex !== -1 && pPermissionIndex !== -1) {
              const allowedPermissions = new Set(); // 使用 Set 來確保權限名稱不重複
              for (let j = 1; j < permissionData.length; j++) {
                const pRow = permissionData[j];
                // 檢查 手機號碼相符 且 權限為 Y
                if (pRow[pPhoneIndex] === payload.key && String(pRow[pPermissionIndex]).toUpperCase() === 'Y') {
                  if (pRow[pSystemFunctionIndex]) { // 確保系統功能欄位有值
                    allowedPermissions.add(pRow[pSystemFunctionIndex].trim());
                  }
                }
              }
              userPermissions = Array.from(allowedPermissions); // 將 Set 轉回陣列
              Logger.log(`[handleLogin] User ${userName} permissions from '權限管理' sheet: [${userPermissions.join(', ')}]`);
            } else {
              Logger.log("[handleLogin] ❌ '權限管理' sheet header misconfigured.");
            }
          }
        } else {
          Logger.log("[handleLogin] ⚠️ '權限管理' sheet not found.");
        }
      } catch (err) {
        Logger.log(`[handleLogin] ❌ Error fetching permissions from '權限管理' sheet: ${err.message}`);
        // 在發生錯誤時，給予空權限，以避免登入流程完全失敗
        userPermissions = [];
      }
      // --- 權限新邏輯結束 ---

      // 如果有提供建案名稱，則驗證建案權限
      if (payload.projectName) {
        const allowedProjects = (row[projectIndex] || '').split(',').map(p => p.trim());
        const requestedProject = payload.projectName;
        Logger.log("[handleLogin] User " + userName + " allowed projects: [" + allowedProjects.join(', ') + "], requested: " + requestedProject);

        if (!allowedProjects.includes(requestedProject)) {
          result = { status: 'no_permission', message: '無此建案權限' };
          loginStatus = '建案權限不足';
          Logger.log("[handleLogin] Permission denied for user: " + userName + " on project: " + requestedProject);
          break;
        }
        Logger.log("[handleLogin] Permission granted for user: " + userName + " on project: " + requestedProject);
        result = {
          status: 'success',
          user: {
            key: row[phoneIndex],
            email: row[emailIndex],
            name: row[nameIndex],
            projectName: payload.projectName,
            permissions: userPermissions // ✅ 使用新邏輯的權限
          }
        };
      } else {
        // 如果未提供建案名稱 (例如，初次登入選擇建案之前)
        Logger.log("[handleLogin] Basic login successful for user: " + userName + ". No specific project selected at login.");
        result = {
          status: 'success',
          user: {
            key: row[phoneIndex],
            email: row[emailIndex],
            name: row[nameIndex],
            projectName: null,
            permissions: userPermissions // ✅ 使用新邏輯的權限
          }
        };
      }
      loginStatus = '成功';
      Logger.log("[handleLogin] Login process completed for user: " + userName + ". Returning user data: " + JSON.stringify(result.user));
      break;
    }
  }

  logLoginRecord(userName || payload.key, loginStatus, payload.userAgent, payload.ip);
  return createJsonResponse(result);
}

        break;
      }
    }

    if (foundRow === -1) {
      Logger.log(`[handleDeletePhotoFromRecord] ⚠️ 找不到對應資料 (key: ${key} 不存在)`);
      
    }

        let filenameToDelete = "";
    const photoNum = photoField.replace(/\D/g, ''); // "1", "2", "3", "4"

    if (photoField.startsWith('photo')) { 
        filenameToDelete = `${key}_缺失照片${photoNum}.jpg`;
    } else if (photoField.startsWith('defectPhoto')) { 
        filenameToDelete = `${key}_檢修照片${photoNum}.jpg`; 
    }

    if (!filenameToDelete) {
        Logger.log(`[handleDeletePhotoFromRecord] ⚠️ 無法確定要刪除的 Drive 文件名，photoField: ${photoField}`);
        return createJsonResponse({
            status: 'warning', // 或 'success' 如果試算表已清除
            message: `試算表欄位已清除，但無法確定 Drive 文件名。`
        });
    }

    Logger.log(`[handleDeletePhotoFromRecord] 準備從 Drive 刪除文件: ${filenameToDelete}`);
    const folder = DriveApp.getFolderById('1js6n9EOFaiN1YrPKIgR3-_d-Gc5b8Lma'); // 確保這是正確的 Folder ID
    const files = folder.getFilesByName(filenameToDelete);
    let fileDeleted = false;

    while (files.hasNext()) {
      const file = files.next();
      file.setTrashed(true);
      fileDeleted = true;
      Logger.log(`[handleDeletePhotoFromRecord] ✅ 已將 Drive 文件 ${file.getName()} (ID: ${file.getId()}) 移至垃圾桶`);
    }

    if (foundRow === -1 && !fileDeleted) {
        return createJsonResponse({ status: 'error', message: `找不到對應資料 (key: ${key}) 且 Drive 中未找到文件 ${filenameToDelete}` });
    }

    return createJsonResponse({
      status: 'success',
      message: (foundRow !== -1 ? '試算表欄位已清除。' : `試算表中未找到 key: ${key}。`) +
               (fileDeleted ? `Drive 文件 ${filenameToDelete} 已刪除。` : (filenameToDelete ? `Drive 中未找到文件 ${filenameToDelete}。` : ''))
    });

  } catch (err) {
    Logger.log('[handleDeletePhotoFromRecord] ❌ 執行時發生錯誤: ' + err.message + ' Stack: ' + err.stack);
    return createJsonResponse({ status: 'error', message: '刪除照片過程中發生錯誤: ' + err.message });
  }
}

    for (let i = 1; i < data.length; i++) {

      const existingToken = data[i][tokenColIndex];
      // const rowProjectName = data[i][projectNameColIndex]; // 如果表中仍有此列

  }

function generateRandomToken(length) {
  let result = '';
  const characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  const charactersLength = characters.length;
  for (let i = 0; i < length; i++) {
    result += characters.charAt(Math.floor(Math.random() * charactersLength));
  }
  return result;
}


function formatDateTime(value) {
  if (value instanceof Date) {
    const yyyy = value.getFullYear();
    const mm = String(value.getMonth() + 1).padStart(2, '0');
    const dd = String(value.getDate()).padStart(2, '0');
    const hh = String(value.getHours()).padStart(2, '0');
    const mi = String(value.getMinutes()).padStart(2, '0');
    return `${yyyy}-${mm}-${dd} ${hh}:${mi}`;
  }
  return value || '';
}

  }
  if (!ssId) {
    Logger.log('[handleConfirmInspection] ❌ Invalid ssId.');
    return createJsonResponse({ status: 'error', message: '系統內部錯誤：無法定位建案資料表 (ssId invalid for confirm_inspection)。' });

  try {

    const projectSpreadsheet = SpreadsheetApp.openById(ssId); // ✅ 使用傳入的 ssId

    const confirmRecordSheetName = '確認驗屋紀錄'; // 將表名定義為常量

    if (!confirmSheet) {

      return createJsonResponse({ status: 'error', message: `系統配置錯誤：找不到 "${confirmRecordSheetName}" 資料表。` });
    }
    confirmSheet.appendRow([now, unitId, owner, phone, email, signatureUrl]);
    Logger.log(`[handleConfirmInspection] ✅ Appended to "${confirmRecordSheetName}" sheet.`);

    // ✅ 更新「驗屋紀錄」表格中的「產權人確認」欄位為 Y
    const inspectionRecordSheetName = '驗屋紀錄'; // 將表名定義為常量
    const recordSheet = projectSpreadsheet.getSheetByName(inspectionRecordSheetName);
    if (!recordSheet) {
      Logger.log(`[handleConfirmInspection] ❌ Sheet "${inspectionRecordSheetName}" not found in ssId: ${ssId}`);
      // 即使這個表找不到，上面的 appendRow 可能已經成功，所以這裡可以考慮返回一個警告或部分成功的消息
      return createJsonResponse({ status: 'warning', message: `確認記錄已寫入，但找不到 "${inspectionRecordSheetName}" 資料表以更新狀態。` });
    }

    const data = recordSheet.getDataRange().getValues();
    if (data.length === 0) {
        Logger.log(`[handleConfirmInspection] ⚠️ Sheet "${inspectionRecordSheetName}" is empty, cannot update status.`);
        return createJsonResponse({ status: 'success', message: '已完成確認紀錄 (但驗屋紀錄表為空，未更新狀態)。' });

    const unitColName = '戶別';
    const ownerConfirmColName = '產權人確認'; // 確認你的欄位名

    const unitIndex = header.indexOf(unitColName);
    const confirmIndex = header.indexOf(ownerConfirmColName);

    if (unitIndex === -1) {

      return createJsonResponse({ status: 'warning', message: `確認記錄已寫入，但 "${inspectionRecordSheetName}" 表缺少 "${unitColName}" 欄。` });

      return createJsonResponse({ status: 'warning', message: `確認記錄已寫入，但 "${inspectionRecordSheetName}" 表缺少 "${ownerConfirmColName}" 欄。` });

      if (data[i][unitIndex] === unitId) {

        // break;

    }

        Logger.log(`[handleConfirmInspection] ⚠️ UnitId "${unitId}" not found in "${inspectionRecordSheetName}" sheet to update status.`);
    }

    return createJsonResponse({ status: 'success', message: '已完成確認紀錄。' + (!updated && data.length > 1 ? ` (但未在驗屋紀錄表中找到戶別 ${unitId} 更新狀態)` : '') });

  } catch (err) {
    Logger.log('[handleConfirmInspection] ❌ Error: ' + err.message + ' Stack: ' + err.stack);
    return createJsonResponse({ status: 'error', message: '寫入確認紀錄時發生錯誤: ' + err.message });

        const projectNames = permissionsString.split(',').map(p => p.trim()).filter(p => p);
        // Format for v-select: { text: 'Name', value: 'Name' }
        projects = projectNames.map(name => ({ text: name, value: name }));
      }
      Logger.log("[handleGetProjectList] User " + userKey + " has permissions: " + permissionsString + ", formatted projects: " + JSON.stringify(projects));
      break;
    }
  }

  if (!foundUser) {
    Logger.log("[handleGetProjectList] User key " + userKey + " not found in 用戶資料.");
    // Return empty list or error, depending on desired behavior
    return createJsonResponse({ status: 'success', projects: [] }); // Or status: 'error', message: 'User not found'
  }

  return createJsonResponse({ status: 'success', projects: projects });

      console.error(`Log sheet "${LOG_SHEET_NAME}" not found in spreadsheet ID "${LOG_SPREADSHEET_ID}". Logging to Logger.log instead.`);
      Logger.log(`[${level}] ${functionName}: ${message} ${details ? JSON.stringify(details) : ''}`);
      return;
    }

    const timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss:SSS');
    const detailsString = details ? JSON.stringify(details) : '';

    // 確保表頭存在，如果不存在則添加
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['Timestamp', 'Level', 'Function', 'Message', 'Details']);
    }

    sheet.appendRow([timestamp, level, functionName, message, detailsString]);

  } catch (e) {
    // 如果寫入工作表失敗，回退到 Logger.log，避免日誌丟失或導致主流程失敗
    console.error(`Failed to write to sheet log: ${e.toString()}. Logging to Logger.log instead.`);
    Logger.log(`[${level} - SHEET_LOG_FALLBACK] ${functionName}: ${message} ${details ? JSON.stringify(details) : ''}`);
  }
}



function formatDriveUrlForEmbed(originalUrl, type = "image") {
  const FUNCTION_NAME = 'formatDriveUrlForEmbed';
  if (!originalUrl || typeof originalUrl !== 'string' || originalUrl.trim() === '') {
    Logger.log(`[${FUNCTION_NAME} - ${type}] Input URL is invalid or empty: '${originalUrl}'`);
    return '';
  }
  if (originalUrl.includes('drive.google.com/uc?export=view&id=')) return originalUrl;
  const patterns = [ /\/file\/d\/([a-zA-Z0-9_-]{25,})\//i, /id=([a-zA-Z0-9_-]{25,})/i, /\/d\/([a-zA-Z0-9_-]{25,})/i, /uc\?id=([a-zA-Z0-9_-]{25,})/i ];
  for (const pattern of patterns) {
    const match = originalUrl.match(pattern);
    if (match && match[1]) {
      const fileId = match[1];
      const convertedUrl = `https://drive.google.com/uc?export=view&id=${fileId}`;
      Logger.log(`[${FUNCTION_NAME} - ${type}] Converted URL: ${convertedUrl} from Original: ${originalUrl} using pattern: ${pattern.toString()}`);
      return convertedUrl;
    }
  }
  Logger.log(`[${FUNCTION_NAME} - ${type}] Could not convert URL: ${originalUrl}. No pattern matched. Returning original URL.`);
  return originalUrl;
}

const SLIDE_TEMPLATE_ID = '1KV2hkyd6axhFQWBoMdCWpGlvl-ZUf-igRNCY_PlW188'; // 你的 Google Slides 模板 ID
const RECORD_DETAIL_TEMPLATE_SLIDE_IDENTIFIER = "驗屋紀錄內頁"; // 用於標識記錄模板幻燈片的文本 (在形狀內或備註中)

/**
 * 從 Google Slides 模板創建包含封面和所有驗屋記錄詳情的 PDF。
 * @param {string} unitId 要生成報告的戶別 ID。
 * @param {boolean} [overwrite=false] 如果目標文件夾中已存在同名 PDF，是否覆蓋。
 * @param {string} ssId 包含建案數據 (戶別資料, 確認驗屋紀錄, 驗屋紀錄) 的 Spreadsheet ID。
 * @return {object} 包含 status 和 url (如果成功) 或 message (如果失敗) 的對象。
 */
function createPdfFromTemplate(unitId, overwrite = false, ssId) {
  const FUNCTION_NAME = 'createPdfFromTemplate_UsingSlides';
  Logger.log(`--- ${FUNCTION_NAME} Start --- Unit: ${unitId}, Overwrite: ${overwrite}, DataSource SsId: ${ssId}`);
  let tempPresentationFileId = null;

  try {
    // --- 1. 獲取數據 ---
    const projectDataSpreadsheet = SpreadsheetApp.openById(ssId);
    const folderSheet = projectDataSpreadsheet.getSheetByName('戶別資料');
    const confirmSheet = projectDataSpreadsheet.getSheetByName('確認驗屋紀錄');
    const inspectionSheet = projectDataSpreadsheet.getSheetByName('驗屋紀錄');
    const pdfLogSheet = projectDataSpreadsheet.getSheetByName('驗屋紀錄PDF');

    if (!folderSheet || !confirmSheet || !inspectionSheet || !pdfLogSheet) {
        const missing = [!folderSheet && "戶別資料",!confirmSheet && "確認驗屋紀錄",!inspectionSheet && "驗屋紀錄",!pdfLogSheet && "驗屋紀錄PDF"].filter(Boolean).join(', ');
        throw new Error(`數據源 Spreadsheet (ID: ${ssId}) 中找不到必要的工作表: ${missing}`);
    }

    const folderDataValues = folderSheet.getDataRange().getValues();
    if (folderDataValues.length < 1) throw new Error('工作表 "戶別資料" 為空。');
    const folderHeader = folderDataValues[0];
    const getFIndex = (label) => { const i = folderHeader.indexOf(label); if (i === -1) throw new Error(`"戶別資料" 表頭找不到: "${label}"`); return i; };
    const houseDataRow = folderDataValues.slice(1).find(r => r[getFIndex('戶別')] === unitId);
    if (!houseDataRow) throw new Error(`在 "戶別資料" 中找不到戶別: ${unitId}`);

    const building = houseDataRow[getFIndex('棟別')] || '';
    // unitId 參數即為戶別
    const parking = houseDataRow[getFIndex('車位')] || '';
    const address = houseDataRow[getFIndex('門牌')] || '';
    const owner = houseDataRow[getFIndex('買方姓名')] || '';
    const phone = houseDataRow[getFIndex('電話')] || '';
    const email = houseDataRow[getFIndex('EMAIL')] || '';
    const appointmentDate = formatSheetDate(houseDataRow[getFIndex('狀態日期')]);
    const appointmentTime = houseDataRow[getFIndex('狀態時段')] || '';
    const targetFolderUrl = houseDataRow[getFIndex('驗屋文件')];

    if (!targetFolderUrl) throw new Error(`戶別 ${unitId} 的 "驗屋文件" 資料夾網址未設定。`);
    const folderIdMatch = targetFolderUrl.match(/[-\w]{25,}/);
    const targetFolderId = folderIdMatch ? folderIdMatch[0] : null;
    if (!targetFolderId) throw new Error(`無法從 "${targetFolderUrl}" 中提取有效的 Google Drive 資料夾 ID。`);
    const outputFolder = DriveApp.getFolderById(targetFolderId);

   let signatureUrl = ''; // 初始化為空
    // const mockSignatureUrl = null; // 在生產函數中，我們不應該依賴 mock

    // 從 "確認驗屋紀錄" 工作表獲取最新的簽名 URL
    const confirmValues = confirmSheet.getDataRange().getValues();
    if (confirmValues.length > 1) { // 確保至少有表頭和一行數據
      const confirmHeader = confirmValues[0];
      const getCIndex = (label) => {
          const i = confirmHeader.indexOf(label);
          // 如果找不到關鍵列，則記錄警告，但不拋出錯誤，因為簽名可能是可選的
          if (i === -1 && (label === '戶別' || label === '簽名圖檔URL' || label === '建立時間')) {
              Logger.log(`[${FUNCTION_NAME}] Warning: "確認驗屋紀錄" 表頭找不到關鍵欄位: "${label}". 將無法獲取簽名。`);
          }
          return i;
      };

      const unitIdColConfirm = getCIndex('戶別');
      const signatureUrlColConfirm = getCIndex('簽名圖檔URL');
      const createdAtColConfirm = getCIndex('建立時間'); // 假設表頭是 '建立時間'

      if (unitIdColConfirm !== -1 && signatureUrlColConfirm !== -1 && createdAtColConfirm !== -1) {
        const confirmDataForUnit = confirmValues
          .slice(1) // 跳過表頭
          .filter(r => r[unitIdColConfirm] === unitId && r[signatureUrlColConfirm] && String(r[signatureUrlColConfirm]).trim() !== '') // 篩選出當前戶別且有簽名URL的記錄
          .map(r => {
            let createdAtDate = null;
            try {
              createdAtDate = new Date(r[createdAtColConfirm]); // 將建立時間轉換為 Date 對象以便排序
              if (isNaN(createdAtDate.valueOf())) createdAtDate = null; // 無效日期處理
            } catch (e) {
              Logger.log(`[${FUNCTION_NAME}] Warning: Could not parse date "${r[createdAtColConfirm]}" for signature record. Unit: ${unitId}`);
            }
            return {
              url: r[signatureUrlColConfirm],
              createdAt: createdAtDate
            };
          })
          .filter(r => r.createdAt); // 只保留有有效建立時間的記錄

        if (confirmDataForUnit.length > 0) {
          // 按建立時間降序排序，最新的在最前面
          confirmDataForUnit.sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime());
          const latestSignatureRecord = confirmDataForUnit[0];
          signatureUrl = formatDriveUrlForEmbed(latestSignatureRecord.url, "signature_from_sheet"); // 使用輔助函數轉換 URL
          Logger.log(`[${FUNCTION_NAME}] Found latest signature for unit ${unitId} from: ${latestSignatureRecord.createdAt.toISOString()}, Original URL: ${latestSignatureRecord.url}, Processed URL: ${signatureUrl}`);
        } else {
          Logger.log(`[${FUNCTION_NAME}] No valid signature records with creation date found for unit ${unitId}.`);
        }
      } else {
        Logger.log(`[${FUNCTION_NAME}] "確認驗屋紀錄" 表缺少必要的列 (戶別, 簽名圖檔URL, 建立時間) 或表頭不匹配，無法獲取簽名。`);
      }
    } else {
      Logger.log(`[${FUNCTION_NAME}] "確認驗屋紀錄" 工作表為空或只有表頭，無法獲取簽名。`);
    }
    
    var records = [];
    const inspectionDataValues = inspectionSheet.getDataRange().getValues();
    if (inspectionDataValues.length > 1) {
        const inspectHeader = inspectionDataValues[0];
        const getIIndex = (label) => { const i = inspectHeader.indexOf(label); if (i === -1) throw new Error(`"驗屋紀錄" 表頭找不到: "${label}"`); return i; };
        const dataRows = inspectionDataValues.slice(1).filter(r => r[getIIndex('戶別')] === unitId && r[getIIndex('已刪除')] !== 'Y');
        records = dataRows.map((row, i) => ({
            index: i + 1, unit: getIIndex('戶別') > -1 ? (row[getIIndex('戶別')] || '') : unitId, ownerFromRecord: getIIndex('產權人') > -1 ? (row[getIIndex('產權人')] || '') : owner, // 避免與封面的 owner 衝突
            inspectionStage: getIIndex('驗屋階段') > -1 ? (row[getIIndex('驗屋階段')] || '') : '',
            inspectionDate: formatSheetDate(getIIndex('驗屋日期') > -1 ? row[getIIndex('驗屋日期')] : ''),
            area: getIIndex('檢查區域') > -1 ? (row[getIIndex('檢查區域')] || '') : '',
            category: getIIndex('分類') > -1 ? (row[getIIndex('分類')] || '') : '',
            subcategory: getIIndex('細項') > -1 ? (row[getIIndex('細項')] || '') : '',
            inspectionStatus: getIIndex('檢查狀態') > -1 ? (row[getIIndex('檢查狀態')] || '') : '',
            description: getIIndex('檢查說明') > -1 ? (row[getIIndex('檢查說明')] || '') : '',
            photo1: formatDriveUrlForEmbed(getIIndex('缺失照片1') > -1 ? row[getIIndex('缺失照片1')] : '', `r${i+1}p1`),
            photo2: formatDriveUrlForEmbed(getIIndex('缺失照片2') > -1 ? row[getIIndex('缺失照片2')] : '', `r${i+1}p2`),
            photo3: formatDriveUrlForEmbed(getIIndex('缺失照片3') > -1 ? row[getIIndex('缺失照片3')] : '', `r${i+1}p3`),
            photo4: formatDriveUrlForEmbed(getIIndex('缺失照片4') > -1 ? row[getIIndex('缺失照片4')] : '', `r${i+1}p4`),
        }));
    }
    Logger.log(`Fetched ${records.length} inspection records for PDF generation.`);

    // --- 2. 複製模板演示文稿 ---
    const nowForFile = new Date(); // 用於文件名和時間戳
    const todayForFile = Utilities.formatDate(nowForFile, Session.getScriptTimeZone(), 'yyyy-MM-dd');
    const tempPresentationName = `Temp_PDF_Slides_${unitId}_${nowForFile.getTime()}`;
    const templateFile = DriveApp.getFileById(SLIDE_TEMPLATE_ID);
    const tempPresentationFile = templateFile.makeCopy(tempPresentationName, DriveApp.getRootFolder());
    tempPresentationFileId = tempPresentationFile.getId();
    const presentation = SlidesApp.openById(tempPresentationFileId);
    Logger.log(`Temporary presentation created: ${tempPresentationName} (ID: ${tempPresentationFileId})`);

    // --- 3. 填充第一張封面幻燈片 ---
    const coverSlide = presentation.getSlides()[0];
    if (!coverSlide) throw new Error("模板中找不到封面幻燈片。");
    const mainInfoPlaceholders = {
      '{{building}}': building, '{{unit}}': unitId, '{{parking}}': parking, '{{address}}': address,
      '{{owner}}': owner, '{{phone}}': phone, '{{email}}': email,
      '{{appointmentDate}}': appointmentDate, '{{appointmentTime}}': appointmentTime
    };
    for (const ph in mainInfoPlaceholders) coverSlide.replaceAllText(ph, mainInfoPlaceholders[ph] || '');
    if (signatureUrl) {
      const shapes = coverSlide.getShapes(); let sigShape = null;
      for (let i = 0; i < shapes.length; i++) { if (shapes[i].getDescription() === SIGNATURE_SHAPE_ALT_TEXT || shapes[i].getTitle() === SIGNATURE_SHAPE_ALT_TEXT) { sigShape = shapes[i]; break; } }
      if (sigShape) {
        const l = sigShape.getLeft(), t = sigShape.getTop(), w = sigShape.getWidth(), h = sigShape.getHeight();
        sigShape.remove();
        try {
          const blob = UrlFetchApp.fetch(signatureUrl).getBlob(); const img = coverSlide.insertImage(blob, l, t, w, h);
          const oW = img.getInherentWidth(), oH = img.getInherentHeight();
          if (oW > 0 && oH > 0) { const sAR = w/h, iAR = oW/oH; let nW,nH; if(iAR > sAR){nW=w;nH=w/iAR;}else{nH=h;nW=h*iAR;} img.setWidth(nW);img.setHeight(nH); img.setLeft(l+(w-nW)/2);img.setTop(t+(h-nH)/2); }
        } catch (e) { Logger.log(`Error inserting signature: ${e.message}`); /* coverSlide.insertTextBox('⚠ 簽名無法載入', l,t,w,h); */ }
      } else { Logger.log(`Shape with AltText "${SIGNATURE_SHAPE_ALT_TEXT}" not found for signature.`); }
    } else { /* 處理無簽名 */ }
    Logger.log("Cover slide processed.");

    // --- 4. 查找“記錄模板幻燈片” ---
    let recordTemplateSlide = null;
    let originalTemplateSlideIndex = -1;
    const slides = presentation.getSlides();
    for (let i = 0; i < slides.length; i++) {
      const shapes = slides[i].getShapes();
      for (let j = 0; j < shapes.length; j++) {
        if (shapes[j].getText && shapes[j].getText().asString().trim() === RECORD_DETAIL_TEMPLATE_SLIDE_IDENTIFIER) {
          recordTemplateSlide = slides[i]; originalTemplateSlideIndex = i; break;
        }
      }
      if (recordTemplateSlide) break;
    }
    if (!recordTemplateSlide && records.length > 0) throw new Error(`模板中找不到標識為 "${RECORD_DETAIL_TEMPLATE_SLIDE_IDENTIFIER}" 的記錄詳情模板幻燈片。`);

    // --- 5. 為每條記錄創建和填充詳情幻燈片 ---
    if (recordTemplateSlide && records.length > 0) {
      records.forEach((record, recIdx) => {
        Logger.log(`Processing record ${record.index} for slide.`);
        const newDetailSlide = recordTemplateSlide.duplicate(); // 新幻燈片會添加到末尾

        newDetailSlide.replaceAllText('{{record_index}}', record.index.toString());
        newDetailSlide.replaceAllText('{{unit}}', record.unit || unitId); // 優先用記錄中的，否則用封面的 unitId
        newDetailSlide.replaceAllText('{{owner}}', record.ownerFromRecord || owner); // 優先用記錄中的，否則用封面的 owner
        newDetailSlide.replaceAllText('{{record_inspectionStage}}', record.inspectionStage || '');
        newDetailSlide.replaceAllText('{{record_inspectionDate}}', record.inspectionDate || '');
        newDetailSlide.replaceAllText('{{record_area}}', record.area || '');
        newDetailSlide.replaceAllText('{{record_category}}', record.category || '');
        newDetailSlide.replaceAllText('{{record_subcategory}}', record.subcategory || '');
        newDetailSlide.replaceAllText('{{record_inspectionStatus}}', record.inspectionStatus || '');
        newDetailSlide.replaceAllText('{{record_description}}', record.description || '');

        const recordPhotos = [record.photo1, record.photo2, record.photo3, record.photo4];
        recordPhotos.forEach((photoUrl, photoIdx) => {
          const photoShapeAltText = `${PHOTO_SHAPE_ALT_TEXT_PREFIX}${photoIdx + 1}`;
          const shapesOnNewSlide = newDetailSlide.getShapes();
          let targetPhotoShape = null;
          for (let s = 0; s < shapesOnNewSlide.length; s++) {
            if (shapesOnNewSlide[s].getDescription() === photoShapeAltText || shapesOnNewSlide[s].getTitle() === photoShapeAltText) {
              targetPhotoShape = shapesOnNewSlide[s]; break;
            }
          }
          if (targetPhotoShape) {
            if (photoUrl) {
              try {
                const imageBlob = UrlFetchApp.fetch(photoUrl).getBlob();
                const l=targetPhotoShape.getLeft(), t=targetPhotoShape.getTop(), w=targetPhotoShape.getWidth(), h=targetPhotoShape.getHeight();
                targetPhotoShape.remove();
                const image = newDetailSlide.insertImage(imageBlob, l, t, w, h);
                const oW=image.getInherentWidth(), oH=image.getInherentHeight();
                if(oW>0 && oH>0){const sAR=w/h, iAR=oW/oH; let nW,nH; if(iAR>sAR){nW=w;nH=w/iAR;}else{nH=h;nW=h*iAR;} image.setWidth(nW);image.setHeight(nH); image.setLeft(l+(w-nW)/2);image.setTop(t+(h-nH)/2);}
              } catch (e) { Logger.log(`Error processing image for ${photoShapeAltText} (Rec ${record.index}): ${e.message}`); try{targetPhotoShape.getText().setText('⚠ 圖無法載入');}catch(e2){} }
            } else { targetPhotoShape.remove(); }
          } else { Logger.log(`Photo shape "${photoShapeAltText}" not found for record ${record.index}.`); }
        });
      });
      if (originalTemplateSlideIndex !== -1 && presentation.getSlides().length > 1) {
        presentation.getSlides()[originalTemplateSlideIndex].remove(); // 刪除原始模板幻燈片
      }
    } else if (records.length === 0 && recordTemplateSlide && presentation.getSlides().length > 1) {
        recordTemplateSlide.remove(); // 如果沒有記錄，也刪除模板幻燈片
    }

    presentation.saveAndClose();
    Logger.log(`All slides processed in temporary presentation.`);

    // --- 6. 導出為 PDF ---
    const pdfFileName = `${unitId}_${owner}_驗屋紀錄_${todayForFile}.pdf`;
    Logger.log(`Attempting to export PDF: ${pdfFileName}`);
    const pdfBlob = DriveApp.getFileById(tempPresentationFileId).getAs('application/pdf');
    pdfBlob.setName(pdfFileName);

    // 檢查 PDF 是否已存在於目標文件夾
    const filesInOutputFolder = outputFolder.getFilesByName(pdfFileName);
    if (filesInOutputFolder.hasNext()) {
      if (!overwrite) {
        const existingPdfUrl = filesInOutputFolder.next().getUrl();
        Logger.log(`PDF "${pdfFileName}" already exists in target folder. Overwrite not set.`);
        // 在 finally 中刪除臨時演示文稿
        return { status: 'exists', message: `${pdfFileName} 已存在，是否覆蓋？`, url: existingPdfUrl };
      }
      Logger.log(`Overwriting existing PDF "${pdfFileName}" in target folder.`);
      while (filesInOutputFolder.hasNext()) filesInOutputFolder.next().setTrashed(true);
    }

    const savedPdfFile = outputFolder.createFile(pdfBlob);
    Logger.log(`SUCCESS: PDF "${pdfFileName}" generated. URL: ${savedPdfFile.getUrl()}`);
    const timezone = Session.getScriptTimeZone();
    const timestamp = Utilities.formatDate(new Date(), timezone, 'yyyy-MM-dd HH:mm:ss');
    pdfLogSheet.appendRow([timestamp, unitId, owner, email, savedPdfFile.getUrl()]);

    return { status: 'success', url: savedPdfFile.getUrl() };

  } catch (e) {
    Logger.log(`ERROR in ${FUNCTION_NAME}: ${e.toString()} \nStack: ${e.stack}`);
    throw e;
  } finally {
    if (tempPresentationFileId) {
      try {
        DriveApp.getFileById(tempPresentationFileId).setTrashed(true);
        Logger.log(`Temporary Google Slides (ID: ${tempPresentationFileId}) deleted.`);
      } catch (deleteError) {
        Logger.log(`Failed to delete temporary Google Slides (ID: ${tempPresentationFileId}): ${deleteError.toString()}`);
      }
    }
    Logger.log(`--- ${FUNCTION_NAME} Test Finished ---`);
  }
}


function handleGetProjectsBySystemPermission(payload) {
  const userKey = payload.key;
  const systemName = payload.systemName; // 例如 "驗屋系統"

  if (!userKey || !systemName) {
    Logger.log('[handleGetProjectsBySystemPermission] ❌ Missing userKey or systemName.');
    return createJsonResponse({ status: 'error', message: '缺少用戶識別碼或系統名稱參數。' });
  }

  Logger.log(`[handleGetProjectsBySystemPermission] UserKey: ${userKey}, SystemName: ${systemName}`);

  try {
    const sheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('權限管理');
    if (!sheet) {
      Logger.log('[handleGetProjectsBySystemPermission] ❌ "權限管理" sheet not found.');
      return createJsonResponse({ status: 'error', message: '系統配置錯誤：找不到權限管理資料表。' });
    }

    const data = sheet.getDataRange().getValues();
    if (data.length < 2) { // 至少要有表頭和一行數據
      Logger.log('[handleGetProjectsBySystemPermission] "權限管理" sheet has no data or only header.');
      return createJsonResponse({ status: 'success', projects: [] });
    }

    const header = data[0];
    const phoneIndex = header.indexOf('手機號碼');
    const projectNameIndex = header.indexOf('建案名稱');
    const systemFunctionIndex = header.indexOf('系統功能');
    const permissionIndex = header.indexOf('權限');

    if (phoneIndex === -1 || projectNameIndex === -1 || systemFunctionIndex === -1 || permissionIndex === -1) {
      Logger.log('[handleGetProjectsBySystemPermission] ❌ "權限管理" sheet header misconfigured.');
      return createJsonResponse({ status: 'error', message: '權限管理資料表表頭配置錯誤。' });
    }

    const allowedProjectNames = new Set(); // 使用 Set 確保建案名稱唯一

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (row[phoneIndex] === userKey &&
          row[systemFunctionIndex] === systemName &&
          String(row[permissionIndex]).toUpperCase() === 'Y') {
        if (row[projectNameIndex]) {
          allowedProjectNames.add(row[projectNameIndex]);
        }
      }
    }

    const projects = Array.from(allowedProjectNames).map(name => ({ text: name, value: name }));
    Logger.log(`[handleGetProjectsBySystemPermission] ✅ Found ${projects.length} projects for ${userKey} in ${systemName}: ${JSON.stringify(projects)}`);
    return createJsonResponse({ status: 'success', projects: projects });

  } catch (err) {
    Logger.log('[handleGetProjectsBySystemPermission] ❌ Error: ' + err.message + ' Stack: ' + err.stack);
    return createJsonResponse({ status: 'error', message: '查詢系統建案權限時發生錯誤: ' + err.message });
  }
}

function getSalesControlSheetId(projectName) {
  Logger.log('[getSalesControlSheetId] 正在查找銷控表ID，建案: "' + projectName + '"');
  try {
    const sheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('建案選單');
    if (!sheet) throw new Error('找不到 "建案選單" 工作表配置。');

    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const nameIndex = headers.indexOf('建案名稱');
    const idIndex = headers.indexOf('銷控SheetID'); 

    if (nameIndex === -1 || idIndex === -1) {
      throw new Error('"建案選單" 表頭配置不正確 (找不到 "建案名稱" 或 "銷控SheetID")。');
    }

    const dataRange = sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn());
    const values = dataRange.getValues();

    for (const row of values) {
      if (row[nameIndex] === projectName) {
        const foundSsId = String(row[idIndex] || '').trim();
        if (!foundSsId) {
          throw new Error(`建案 "${projectName}" 的銷控SheetID未設定或為空。`);
        }
        Logger.log(`[getSalesControlSheetId] ✅ 找到匹配! SheetID: "${foundSsId}"`);
        return foundSsId;
      }
    }

    throw new Error(`在 "建案選單" 中找不到建案名稱：${projectName}`);
  } catch (err) {
    Logger.log('[getSalesControlSheetId] 內部錯誤: ' + err.message);
    throw err;
  }
}


// 輔助函數：將工作表數據轉換為物件陣列
function sheetDataToObjects(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];
  const header = values[0];
  return values.slice(1).map(row => {
    const obj = {};
    header.forEach((col, index) => {
      if (col) obj[col] = row[index];
    });
    return obj;
  });
}

// 在你的 GAS 腳本中添加這個輔助函數
function getEmbeddableDriveInfo(file) {
  const fileId = file.getId();
  const name = file.getName();
  const mimeType = file.getMimeType();
  let embedUrl = '';
  let type = 'other';

  if (mimeType.startsWith('image/')) {
    // 圖片格式
    type = 'image';
    embedUrl = `https://drive.google.com/uc?export=view&id=${fileId}`;
  } else if (mimeType === MimeType.PDF) {
    // PDF 格式
    type = 'pdf';
    embedUrl = `https://docs.google.com/gview?url=https://drive.google.com/uc?id=${fileId}&embedded=true`;
  } else {
    // 其他文件類型，提供一個標準的查看鏈接
    embedUrl = file.getUrl();
  }

  return {
    name: name,
    embedUrl: embedUrl,
    type: type,
    originalUrl: file.getUrl() // 保留原始鏈接以備用
  };
}

/**
 * 獲取報價人員列表。
 * - 如果用戶有編輯權限，則從建案的 '銷售人員' 工作表獲取完整列表（包含姓名和電話）。
 * - 如果沒有，則列表只包含用戶本人。
 * @param {object} payload 包含 projectName 和 key (currentUserKey) 的請求體。
 * @returns {ContentService.TextOutput} 包含人員列表和編輯權限的 JSON 響應。
 */
function handleGetQuotePersonnelList(payload) {
  const functionName = 'handleGetQuotePersonnelList';
  writeToSheetLog(`--- ${functionName} 開始 ---`, { payload });

  const { projectName, key: currentUserKey } = payload;

  if (!projectName || !currentUserKey) {
    const errorMsg = '獲取報價人員時缺少建案名稱或當前用戶資訊。';
    writeToSheetLog(`❌ ${functionName} 參數錯誤`, { error: errorMsg });
    return createJsonResponse({ status: 'error', message: errorMsg });
  }

  try {
    let personnelList = [];
    let currentUserCanEdit = false;

    // --- 步驟 1: 判斷用戶是否有編輯權限 (邏輯不變) ---
    const permissionSheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('權限管理');
    if (permissionSheet) {
      const permissionData = permissionSheet.getDataRange().getValues();
      const permissionHeader = permissionData[0];
      const pPhoneIndex = permissionHeader.indexOf('手機號碼');
      const pProjectNameIndex = permissionHeader.indexOf('建案名稱');
      const pSystemFunctionIndex = permissionHeader.indexOf('系統功能');
      const pPermissionIndex = permissionHeader.indexOf('權限');

      if (pPhoneIndex !== -1 && pProjectNameIndex !== -1 && pSystemFunctionIndex !== -1 && pPermissionIndex !== -1) {
        for (let i = 1; i < permissionData.length; i++) {
          const pRow = permissionData[i];
          if (pRow[pPhoneIndex] === currentUserKey &&
              pRow[pSystemFunctionIndex] === '報價系統銷售選單' &&
              pRow[pProjectNameIndex] === projectName &&
              String(pRow[pPermissionIndex]).toUpperCase() === 'Y') {
            currentUserCanEdit = true;
            Logger.log(`[${functionName}] ✅ Permission GRANTED for user ${currentUserKey} to edit sales menu in project ${projectName}.`);
            break;
          }
        }
      } else {
        Logger.log(`[${functionName}] "權限管理" 表頭配置錯誤。`);
      }
    } else {
      Logger.log(`[${functionName}] ⚠️ 找不到 "權限管理" 工作表，無法判斷編輯權限。`);
    }

    // --- 步驟 2: 根據編輯權限，決定人員列表的來源 ---
    if (currentUserCanEdit) {
      // **最終新邏輯**: 有權限，直接從建案的「銷售人員」工作表同時讀取姓名與電話
      Logger.log(`[${functionName}] 用戶有編輯權限，從 "${projectName}" 的 "銷售人員" 表獲取列表。`);
      try {
        const salesSsId = getSalesControlSheetId(projectName);
        const salesSpreadsheet = SpreadsheetApp.openById(salesSsId);
        const salesPersonnelSheet = salesSpreadsheet.getSheetByName('銷售人員');
        
        if (salesPersonnelSheet) {
          const salesData = salesPersonnelSheet.getDataRange().getValues();
          if (salesData.length > 1) {
            const salesHeader = salesData[0];
            // 同時取得 "銷售人員" 和 "銷售電話" 的欄位索引
            const salesPersonColIndex = salesHeader.indexOf('銷售人員');
            const salesPhoneColIndex = salesHeader.indexOf('銷售電話'); // ✅ 新增

            if (salesPersonColIndex !== -1 && salesPhoneColIndex !== -1) { // ✅ 確保兩個欄位都存在
              for(let i = 1; i < salesData.length; i++) {
                const row = salesData[i];
                const personName = row[salesPersonColIndex];
                const personPhone = row[salesPhoneColIndex]; // ✅ 讀取電話

                if(personName && String(personName).trim() !== '') {
                   // 直接組合姓名和電話
                  personnelList.push({ 
                    name: String(personName).trim(), 
                    phone: String(personPhone || '').trim() // ✅ 如果電話為空，給予空字串
                  });
                }
              }
               Logger.log(`[${functionName}] 從銷售人員列表成功獲取 ${personnelList.length} 位人員。`);
            } else {
               Logger.log(`[${functionName}] "銷售人員" 工作表中找不到 "銷售人員" 或 "銷售電話" 欄位。`);
            }
          }
        } else {
          Logger.log(`[${functionName}] 在建案 "${projectName}" 中找不到 "銷售人員" 工作表。`);
        }
      } catch (e) {
         Logger.log(`[${functionName}] 讀取銷售人員列表時出錯: ${e.message}`);
         personnelList = [];
      }

    } else {
      // **無權限**: 列表只包含用戶自己 (邏輯不變)
      Logger.log(`[${functionName}] 用戶無編輯權限，列表預設為用戶本人。`);
      const userSheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('用戶資料');
      if (userSheet) {
        const userData = userSheet.getDataRange().getValues();
        const userHeader = userData[0];
        const nameIndex = userHeader.indexOf('NAME');
        const phoneIndex = userHeader.indexOf('手機號碼');

        if (nameIndex !== -1 && phoneIndex !== -1) {
          for (let i = 1; i < userData.length; i++) {
            if (userData[i][phoneIndex] === currentUserKey) {
              const currentUserName = userData[i][nameIndex];
              personnelList.push({ name: currentUserName, phone: currentUserKey });
              break;
            }
          }
        }
      }
    }
    
    // **備援機制**: 如果在任何情況下列表都是空的，至少要確保當前使用者在裡面
    if (personnelList.length === 0) {
        Logger.log(`[${functionName}] 經過所有邏輯後，人員列表仍為空。再次嘗試將當前用戶作為預設值。`);
        const userSheet = SpreadsheetApp.openById(FIXED_ACCOUNT_SHEET_ID).getSheetByName('用戶資料');
        const userData = userSheet.getDataRange().getValues();
        const userHeader = userData[0];
        const nameIndex = userHeader.indexOf('NAME');
        const phoneIndex = userHeader.indexOf('手機號碼');
        for (let i = 1; i < userData.length; i++) {
            if (userData[i][phoneIndex] === currentUserKey) {
                personnelList.push({ name: userData[i][nameIndex], phone: currentUserKey });
                break;
            }
        }
    }

    writeToSheetLog(`--- ${functionName} 執行完畢 ---`, {
      foundPersonnelCount: personnelList.length,
      currentUserCanEdit: currentUserCanEdit,
      finalList: JSON.stringify(personnelList)
    });

    return createJsonResponse({
      status: 'success',
      data: {
        personnelList: personnelList,
        canEdit: currentUserCanEdit
      }
    });

  } catch (err) {
    writeToSheetLog(`❌ ${functionName} 捕獲到嚴重錯誤`, {
      error: err.message,
      stack: err.stack
    });
    return createJsonResponse({ status: 'error', message: `獲取報價人員列表時發生嚴重錯誤: ${err.message}` });
  }
}

      const joinValues = (key) => items.map(item => item[key] || '').join('\n');
      
      logSheet.appendRow([

      Logger.log('找不到名為 "報價紀錄" 的工作表，跳過寫入日誌。');

    // --- 處理 Slide ---
    const today = Utilities.formatDate(new Date(), "Asia/Taipei", "yyyyMMdd-HHMMSS");
    const tempSlideName = `${projectName}_報價單_${today}_${new Date().getTime()}`;
    const tempSlideFile = DriveApp.getFileById(slideTemplateId).makeCopy(tempSlideName);
    tempSlideFileId = tempSlideFile.getId();
    
    const presentation = SlidesApp.openById(tempSlideFileId);
    const slide = presentation.getSlides()[0]; // 假設所有內容都在第一頁

    // 替換通用資訊
    slide.replaceAllText('{{建案名稱}}', projectName || '');
    slide.replaceAllText('{{報價人員}}', personnelName || '');
    slide.replaceAllText('{{連絡電話}}', personnelPhone || '');
    slide.replaceAllText('{{報價日期}}', Utilities.formatDate(new Date(), "Asia/Taipei", "yyyy/MM/dd"));

    const MAX_ITEMS = 5;

    // 填入項目資料
    items.forEach((item, index) => {
      const i = index + 1;
      const fields = ['戶別', '是否首購', '房屋總面積', '房屋總價', '單價', '車位編號', '車位價格', '配套價', '總價', '訂金', '訂金金額', '簽約金%', '簽約金金額', '工程期款%', '工程期款金額', '使照取得%', '使照取得金額', '交屋款%', '交屋款金額', '銀行貸款%', '銀行貸款金額'];

        slide.replaceAllText(`{{${field}_${i}}}`, item[field] !== undefined ? item[field] : '');
      });

    // 清理多餘的佔位符

      const k = i + 1;
      const fields = ['戶別', '是否首購', '房屋總面積', '房屋總價', '單價', '車位編號', '車位價格', '配套價', '總價', '訂金', '訂金金額', '簽約金%', '簽約金金額', '工程期款%', '工程期款金額', '使照取得%', '使照取得金額', '交屋款%', '交屋款金額', '銀行貸款%', '銀行貸款金額'];
      fields.forEach(field => {
        slide.replaceAllText(`{{${field}_${k}}}`, '');
      });
    }

    presentation.saveAndClose();

    // --- 產出 PDF 並儲存 ---
    const pdfBlob = tempSlideFile.getAs('application/pdf');
    const pdfFileName = `${personnelName}_${projectName}_報價單_${today}.pdf`;
    const targetFolder = DriveApp.getFolderById(targetFolderId);
    
    // 如果已存在同名檔案，先刪除舊檔

    while (existingFiles.hasNext()) {
       existingFiles.next().setTrashed(true);
    }
    
    const pdfFile = targetFolder.createFile(pdfBlob).setName(pdfFileName);
    pdfFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

    Logger.log(`✅ PDF 產出成功: ${pdfFile.getUrl()}`);
    return createJsonResponse({ status: 'success', url: pdfFile.getUrl() });

  } catch (err) {
    Logger.log(`❌ ${functionName} 執行錯誤: ${err.message} \nStack: ${err.stack}`);
    return createJsonResponse({ status: 'error', message: `產生報價單失敗: ${err.message}` });
  } finally {
    // --- 清理臨時檔案 ---
    if (tempSlideFileId) {
      try {
        DriveApp.getFileById(tempSlideFileId).setTrashed(true);
      } catch (e) {}
    }
  }
}


function handleUpdateSalesData(payload) {
  const functionName = 'handleUpdateSalesData';
  Logger.log(`--- ${functionName} 開始 --- Payload: ${JSON.stringify(payload)}`);
  
  const { projectName, unitId, salesData, buyerData, parkingData } = payload; // parkingData 是最終選定的車位列表
  if (!projectName || !unitId) {
    Logger.log(`[${functionName}] ❌ 請求的資料不完整(缺少建案或戶別)。`);
    return createJsonResponse({ status: 'error', message: '請求的資料不完整(缺少建案或戶別)。' });
  }

  try {
    const ssId = getSalesControlSheetId(projectName);
    const spreadsheet = SpreadsheetApp.openById(ssId);

    // --- 1. 更新「銷控」工作表 ---
    const salesSheet = spreadsheet.getSheetByName('銷控');
    if (salesSheet && salesData) {
      const salesHeader = salesSheet.getRange(1, 1, 1, salesSheet.getLastColumn()).getValues()[0];
      const unitIdColSales = salesHeader.indexOf('戶別');
      if (unitIdColSales === -1) {
        Logger.log(`[${functionName}] ❌ 銷控表中找不到 "戶別" 欄位`);
        throw new Error('銷控表中找不到 "戶別" 欄位');
      }
      
      const lastRowSales = salesSheet.getLastRow();
      if (lastRowSales > 1) {
        const dataRangeSales = salesSheet.getRange(2, 1, lastRowSales - 1, salesSheet.getLastColumn());
        const salesValues = dataRangeSales.getValues();

        for (let i = 0; i < salesValues.length; i++) {
          if (salesValues[i][unitIdColSales] === unitId) {
            const rowIdx = i + 2;
            Logger.log(`[${functionName}] 更新銷控表，戶別: ${unitId}, 列: ${rowIdx}`);
            
            if (salesData['銷控後台狀態'] && String(salesData['銷控後台狀態']).trim() !== '') {
              salesData['銷控狀態'] = '已售';
            }
            
            Object.keys(salesData).forEach(key => {
              const colIdx = salesHeader.indexOf(key);
              if (colIdx !== -1) {
                let value = salesData[key];
                if (['小訂日期', '補足日期', '簽約日期'].includes(key) && value) {
                    try {
                        value = Utilities.formatDate(new Date(value), "Asia/Taipei", "yyyy/MM/dd");
                    } catch (e) { Logger.log(`[${functionName}] 日期轉換錯誤 for key ${key}, value ${salesData[key]}: ${e.message}`); }
                }
                salesSheet.getRange(rowIdx, colIdx + 1).setValue(value);
              }
            });
            break;
          }
        }
      } else {
        Logger.log(`[${functionName}] 銷控表為空或只有表頭，跳過更新。`);
      }
    } else {
      Logger.log(`[${functionName}] 找不到銷控表或 salesData 為空，跳過銷控表更新。`);
    }
    
    // --- 2. 更新「買方資料」工作表 ---
    const buyerSheet = spreadsheet.getSheetByName('買方資料');
    if (buyerSheet && buyerData) {
        const buyerHeader = buyerSheet.getRange(1, 1, 1, buyerSheet.getLastColumn()).getValues()[0];
        const unitIdColBuyer = buyerHeader.indexOf('戶別');
        if(unitIdColBuyer === -1) {
          Logger.log(`[${functionName}] ❌ 買方資料表中找不到 "戶別" 欄位`);
          throw new Error('買方資料表中找不到 "戶別" 欄位');
        }
        
        const lastRowBuyer = buyerSheet.getLastRow();
        let buyerRowFound = false;
        
        if (lastRowBuyer > 1) {
            const dataRangeBuyer = buyerSheet.getRange(2, 1, lastRowBuyer - 1, buyerSheet.getLastColumn());
            const buyerValues = dataRangeBuyer.getValues();
            for (let i = 0; i < buyerValues.length; i++) {
                if (buyerValues[i] && buyerValues[i][unitIdColBuyer] === unitId) {
                    const rowIdx = i + 2;
                    Logger.log(`[${functionName}] 更新買方資料表，戶別: ${unitId}, 列: ${rowIdx}`);
                    Object.keys(buyerData).forEach(key => {
                        const colIdx = buyerHeader.indexOf(key);
                        if (colIdx !== -1) {
                            let value = buyerData[key];
                            if (key === '出生年月日' && value) {
                                try { value = Utilities.formatDate(new Date(value), "Asia/Taipei", "yyyy/MM/dd"); } catch (e) { Logger.log(`[${functionName}] 出生年月日轉換錯誤 for value ${buyerData[key]}: ${e.message}`);}
                            }
                            buyerSheet.getRange(rowIdx, colIdx + 1).setValue(value);
                        }
                    });
                    buyerRowFound = true;
                    break;
                }
            }
        }

        if (!buyerRowFound) {
            Logger.log(`[${functionName}] 在買方資料表中未找到戶別 ${unitId}，準備新增一列。`);
            const newRow = buyerHeader.map(header => {
                let value = buyerData[header] !== undefined ? buyerData[header] : (header === '戶別' ? unitId : '');
                if (header === '出生年月日' && value) {
                    try { value = Utilities.formatDate(new Date(value), "Asia/Taipei", "yyyy/MM/dd"); } catch(e) { Logger.log(`[${functionName}] 新增買方-出生年月日轉換錯誤 for value ${value}: ${e.message}`);}
                }
                return value;
            });
            buyerSheet.appendRow(newRow);
            Logger.log(`[${functionName}] 已新增買方資料列 for ${unitId}。`);
        }
    } else {
      Logger.log(`[${functionName}] 找不到買方資料表或 buyerData 為空，跳過買方資料表更新。`);
    }

    // --- 3. 更新「車位」工作表 ---
    const parkingSheet = spreadsheet.getSheetByName('車位');
    if (parkingSheet) {
        const parkingHeader = parkingSheet.getRange(1, 1, 1, parkingSheet.getLastColumn()).getValues()[0];
        const parkingIdCol = parkingHeader.indexOf('車位編號');
        const unitColParking = parkingHeader.indexOf('購買戶別');
        const statusColParking = parkingHeader.indexOf('銷控狀態');
        const backendStatusColParking = parkingHeader.indexOf('銷控後台狀態');
        const priceColParking = parkingHeader.indexOf('車位成交價');

        if (parkingIdCol !== -1 && unitColParking !== -1) {
            const lastRowParking = parkingSheet.getLastRow();
            let allParkingValues = [];
            if (lastRowParking > 1) {
                allParkingValues = parkingSheet.getRange(2, 1, lastRowParking - 1, parkingSheet.getLastColumn()).getValues();
            }
            
            const finalSelectedParkingIds = (parkingData || []).map(p => p['車位編號']);
            Logger.log(`[${functionName}] 戶別 ${unitId} 最終持有車位: ${JSON.stringify(finalSelectedParkingIds)}`);

            // 遍歷車位表，處理需要取消關聯的車位
            for (let i = 0; i < allParkingValues.length; i++) {
                const rowParking = allParkingValues[i];
                const currentParkingIdInSheet = rowParking[parkingIdCol];
                const currentUnitIdInSheet = rowParking[unitColParking];

                if (currentUnitIdInSheet === unitId && !finalSelectedParkingIds.includes(currentParkingIdInSheet)) {
                    const rowIdxToClear = i + 2;
                    Logger.log(`[${functionName}] 準備清除車位 ${currentParkingIdInSheet} (原屬戶別 ${unitId}) 的關聯。位於列 ${rowIdxToClear}`);
                    if (unitColParking !== -1) parkingSheet.getRange(rowIdxToClear, unitColParking + 1).setValue('');
                    if (statusColParking !== -1) parkingSheet.getRange(rowIdxToClear, statusColParking + 1).setValue(''); 
                    if (backendStatusColParking !== -1) parkingSheet.getRange(rowIdxToClear, backendStatusColParking + 1).setValue('');
                    if (priceColParking !== -1) parkingSheet.getRange(rowIdxToClear, priceColParking + 1).setValue('');
                    Logger.log(`[${functionName}] 已清除車位 ${currentParkingIdInSheet} 與戶別 ${unitId} 的關聯。`);
                }
            }

            // 更新 payload.parkingData 中指定的車位
            if (parkingData && parkingData.length > 0) {
                parkingData.forEach(p_item => {
                    let foundInSheet = false;
                    for (let i = 0; i < allParkingValues.length; i++) {
                        if (allParkingValues[i][parkingIdCol] === p_item['車位編號']) {
                            const rowIdxToUpdate = i + 2;
                            Logger.log(`[${functionName}] 準備更新車位 ${p_item['車位編號']} 給戶別 ${unitId}。位於列 ${rowIdxToUpdate}`);
                            if (statusColParking !== -1) parkingSheet.getRange(rowIdxToUpdate, statusColParking + 1).setValue('已售');
                            
                            let backendStatusToSet = '已售'; // Default backend status
                            if (salesData && salesData['銷控後台狀態'] !== undefined && String(salesData['銷控後台狀態']).trim() !== '') {
                                backendStatusToSet = salesData['銷控後台狀態'];
                            }
                            if (backendStatusColParking !== -1) {
                                parkingSheet.getRange(rowIdxToUpdate, backendStatusColParking + 1).setValue(backendStatusToSet);
                            }

                            if (unitColParking !== -1) parkingSheet.getRange(rowIdxToUpdate, unitColParking + 1).setValue(unitId);
                            if (priceColParking !== -1) parkingSheet.getRange(rowIdxToUpdate, priceColParking + 1).setValue(p_item['車位成交價']);
                            foundInSheet = true;
                            break;
                        }
                    }
                    if (!foundInSheet) {
                        Logger.log(`[${functionName}] 警告：車位 ${p_item['車位編號']} 在「車位」表中未找到，無法更新其為戶別 ${unitId} 持有。`);
                    }
                });
            }
        } else {
            Logger.log(`[${functionName}] 「車位」表中缺少 '車位編號' (索引 ${parkingIdCol}) 或 '購買戶別' (索引 ${unitColParking}) 欄位，跳過車位更新。`);
        }
    } else {
        Logger.log(`[${functionName}] 找不到「車位」工作表，跳過車位更新。`);
    }
    
    SpreadsheetApp.flush();
    Logger.log(`[${functionName}] --- 資料更新成功 ---`);
    return createJsonResponse({ status: 'success', message: '資料更新成功！' });

  } catch (err) {
    Logger.log(`[${functionName}] ❌ 執行錯誤: ${err.message} \nStack: ${err.stack}`);
    return createJsonResponse({ status: 'error', message: `儲存銷控資料失敗: ${err.message}` });
  }
}
